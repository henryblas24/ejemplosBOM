BOM- Browser Object Model

  • Window

    • Ejemplos :

    window.prompt('Hola') ;
    window.alert('Este es una alerta') ;
    window.confirm('Confirmar') ;
    window.open('https://www.google.com') ;

  • Document

    • Ejemplos :

    const link = document.querySelector('a');
    console.log(link);

    const link2 = document.querySelectorAll('nav ul li');
    console.log(link2);

  • History

    • Ejemplos :

    window.history.go(1);
    window.history.back();
    window.history.forward();

    function openPrev() {
    window.history.forward();
    }

  • Location

    • Ejemplos :

    const value1 = document.location.href;
    console.log(value1);

    const value2 = document.location.protocol;
    console.log(value2);

    const value3 = document.location.port;
    console.log(value3);

    console.log(document.location.origin);
    console.log(document.location.pathName);

  • Navigator

    • Ejemplos :

    console.log(window.navigator);

    console.log(window.navigator.appName);

    console.log(window.navigator.version);

    console.log(window.navigator.online);

    console.log(window.navigator.language);

  • Screen

    • Ejemplos :

    const size = window.screen

    console.log(size.widht);

    console.log(size.height);
